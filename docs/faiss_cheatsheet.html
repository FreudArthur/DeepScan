Code minimal complet
import faiss
import numpy as np

d = 128           # dimension des vecteurs
n = 10000         # nombre de vecteurs

# FAISS exige float32 C-contiguous
xb = np.random.randn(n, d).astype('float32')

# Créer + peupler l'index
index = faiss.IndexFlatL2(d)
index.add(xb)

# Recherche des 5 plus proches voisins
xq = np.random.randn(3, d).astype('float32')
D, I = index.search(xq, k=5)
# D: distances (n_queries, k)  I: indices (n_queries, k)
⚡ Toujours float32 et C-contiguous. Si erreur → np.ascontiguousarray(x.astype('float32'))
Choisir son index
IndexUse caseSpeedRAMPrécision
IndexFlatL2Exact search, baseline, <1M vecslentélevée100%
IndexFlatIPCosine sim (avec normalisation)lentélevée100%
IndexIVFFlatANN rapide, >100k vecsrapideélevée~98%
IndexIVFPQMillions de vecs, RAM limitéetrès rapidefaible~95%
IndexHNSWANN haute précision, no trainingrapideélevée~99%
IndexPQCompression max, pas d'IVFmoyentrès faible~90%
🏆 Rule of thumb : <100k → FlatL2 | 100k–10M → IVFFlat ou IVFPQ | >10M → IVFPQ + GPU
Création des index clés
IVFFlat besoin train
# nlist = nb de clusters (√n)
quantizer = faiss.IndexFlatL2(d)
index = faiss.IndexIVFFlat(
    quantizer, d, nlist=100
)
index.train(xb)
index.add(xb)
index.nprobe = 10  # clusters sondés
IVFPQ besoin train
# m=nb sous-vecteurs, bits/subvec
index = faiss.IndexIVFPQ(
    quantizer, d,
    nlist=100, m=8, nbits=8
)
index.train(xb)
index.add(xb)
index.nprobe = 10
HNSW no training
# M = nb de liens par nœud
index = faiss.IndexHNSWFlat(
    d, M=32
)
index.add(xb)  # pas de train()
Cosine similarity
# Normaliser → L2 = cosine
faiss.normalize_L2(xb)
faiss.normalize_L2(xq)
index = faiss.IndexFlatIP(d)
index.add(xb)
String factory (notation compacte)
# Équivalent IVFPQ avec OPQ preprocessing
index = faiss.index_factory(
    d, "OPQ16,IVF256,PQ16", faiss.METRIC_L2
)
# Autres exemples : "Flat", "IVF100,Flat", "HNSW32"
Search
# k plus proches voisins
D, I = index.search(xq, k=10)
# D: (n_queries, k) distances
# I: (n_queries, k) indices (-1 si pas trouvé)

# Search avec range (seuil de distance)
lims, D, I = index.range_search(xq, thresh=0.5)
# lims[i]:lims[i+1] = slice des résultats de la requête i
IDs personnalisés
IndexIDMap wrapper
# Mapper tes propres IDs (ex: DB ids)
base_index = faiss.IndexFlatL2(d)
index = faiss.IndexIDMap(base_index)

my_ids = np.array([100, 200, 300], dtype='int64')
index.add_with_ids(xb[:3], my_ids)

D, I = index.search(xq, k=2)
# I contient tes IDs custom (100, 200...)
Sauvegarde / chargement
# Sauvegarder
faiss.write_index(index, "my_index.faiss")

# Charger
index = faiss.read_index("my_index.faiss")
Supprimer des vecteurs
# Suppression par IDs (besoin IndexIDMap)
ids_to_remove = np.array([200], dtype='int64')
index.remove_ids(faiss.IDSelectorArray(ids_to_remove))

# IndexFlatL2 ne supporte pas remove → reconstruire
⚠ Les index Flat ne supportent pas la suppression directe. Utiliser IDMap ou reconstruire.
Infos sur l'index
index.ntotal      # nb de vecteurs stockés
index.d           # dimension
index.is_trained  # bool
faiss.vector_to_array(index.xb)  # récupérer vecs (FlatL2)
CPU → GPU
# Convertir un index CPU en GPU
res = faiss.StandardGpuResources()
index_gpu = faiss.index_cpu_to_gpu(
    res, gpu_id=0, index=index_cpu
)

# Multi-GPU
index_gpu = faiss.index_cpu_to_all_gpus(index_cpu)
GPU → CPU (pour sauvegarder)
index_cpu = faiss.index_gpu_to_cpu(index_gpu)
faiss.write_index(index_cpu, "saved.faiss")

RAG / Semantic search
Pattern complet embeddings NLP
from sentence_transformers import SentenceTransformer
import faiss, numpy as np

model = SentenceTransformer('all-MiniLM-L6-v2')
docs = ["Hello world", "Faiss is fast", "AI rocks"]

# Encoder les documents
embeddings = model.encode(docs).astype('float32')
d = embeddings.shape[1]

# Créer index + ajouter
index = faiss.IndexFlatIP(d)
faiss.normalize_L2(embeddings)
index.add(embeddings)

# Query
q = model.encode(["fast search"]).astype('float32')
faiss.normalize_L2(q)
D, I = index.search(q, k=2)
# Résultats : [docs[i] for i in I[0]]
Batch search efficace
# Toujours passer un batch plutôt que des boucles
# ✗ Lent
for q in queries:
    D, I = index.search(q[np.newaxis], k)

# ✓ Rapide (vectorisé)
queries = queries.astype('float32')
D, I = index.search(queries, k)
Reconstruction de vecteur
# Récupérer un vecteur par son index (FlatL2 seulement exact)
vec = index.reconstruct(42)

# Plusieurs en une fois
vecs = index.reconstruct_n(start=0, ni=100)
